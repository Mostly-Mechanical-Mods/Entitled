entitled_convert_culture_all = {
	scope = title
	effect = {
		every_in_de_jure_hierarchy = {
			limit = { tier = tier_county }
			continue = { tier > tier_county }
			if = {
				limit = { NOT = { has_variable = entitled_culture }}
				set_variable = { name = entitled_culture value = culture }
			}
			set_county_culture = global_var:entitled_pinned_culture
		}
		every_in_de_jure_hierarchy = {
			limit = {
				tier <= tier_county
				holder.capital_county ?= this
			}
			holder = {
				if = {
					limit = { NOT = { has_variable = entitled_culture }}
					set_variable = { name = entitled_culture value = culture }
				}
				set_culture = global_var:entitled_pinned_culture
				every_courtier = {
					if = {
						limit = { NOT = { has_variable = entitled_culture }}
						set_variable = { name = entitled_culture value = culture }
					}
					set_culture = global_var:entitled_pinned_culture
				}
			}
		}
		set_global_variable = refresh_culture_mapmode
	}
}

	scope = title
	effect = {
		every_in_de_jure_hierarchy = {
			limit = { tier <= tier_county }
			if = {
				limit = {
					tier = tier_county
					has_variable = entitled_culture
					NOT = { culture = var:entitled_culture }
				}
				set_county_culture = var:entitled_culture
				remove_variable = entitled_culture
			}
			if = {
				limit = { holder.capital_county ?= this }
				holder = {
					if = {

							has_variable = entitled_culture

						}

					}

							NOT = { culture = var:entitled_culture }
						}
						set_culture = var:entitled_culture
						remove_variable = entitled_culture
					}
				}
			}
		}

	}
